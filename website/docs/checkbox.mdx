## 演示

### 简单示例

```jsx
export function Demo1() {
  const [checked, changeChecked] = React.useState(false);
  return (
    <Checkbox checked={checked} onChange={changeChecked}>
      Apple
    </Checkbox>
  );
}
```

export function Demo1() {
  const [checked, changeChecked] = React.useState(false);
  return (
    <>
      <Checkbox checked={checked} onChange={changeChecked}>
        Apple
      </Checkbox>
      <div>{String(checked)}</div>
    </>
  );
}

<Demo1 />

### 组

```jsx
export function Demo2() {
  const [value, changeValue] = React.useState([]);
  return (
    <Checkbox.Group value={['apple', 'oneplus']} onChange={changeValue}>
      <Checkbox value={['apple', 'smartisan', 'oneplus']}>All</Checkbox>
      <Checkbox value="apple">Apple</Checkbox>
      <Checkbox value="smartisan">Smartisan</Checkbox>
      <Checkbox value="oneplus">OnePlus</Checkbox>
    </Checkbox.Group>
  );
}
```

export function Demo2() {
  const [value, changeValue] = React.useState([]);
  return (
    <>
      <Checkbox.Group value={value} onChange={changeValue}>
        <Checkbox value={['apple', 'smartisan', 'oneplus']}>All</Checkbox>
        <Checkbox value="apple">Apple</Checkbox>
        <Checkbox value="smartisan">Smartisan</Checkbox>
        <Checkbox value="oneplus">OnePlus</Checkbox>
      </Checkbox.Group>
      <div>[{value.join(', ')}]</div>
    </>
  );
}

<Demo2 />

## API

### Checkbox Props

| 参数 | 类型 | 默认值 | 说明 |
|-----|------|-------|-----|
| checked | `boolean` | - | 是否选中 |
| defaultChecked | `boolean` | `false` | 是否默认选中 |
| value | `any` | - | 值 |
| disabled | `boolean` | `false` | 禁用 |
| onChange | `(checked: boolean, target: CheckboxProps) => void` | - | 变更事件 |
| className | `string` | - | 类名 |
| style | `React.CSSProperties` | - | 样式 |

### Checkbox.Group Props

| 参数 | 类型 | 默认值 | 说明 |
|-----|------|-------|-----|
| value | `any[]` | - | 值数组 |
| defaultValue | `any[]` | `[]` | 默认的值数组 |
| disabled | `boolean` | `false` | 禁用 |
| onChange | `(value: any[], target: CheckboxGroupProps) => void` | - | 变更事件 |
| className | `string` | - | 类名 |
| style | `React.CSSProperties` | - | 样式 |
