## 演示

### 简单示例

```jsx
export function Demo1(params) {
  const [text, setText] = React.useState('');
  return (
    <div style={{ position: 'relative' }}>
      <div
        style={{
          position: 'absolute',
          top: '0',
          left: '0',
          width: '100%',
          backgroundColor: 'rgba(255,255,255,.7)',
          lineHeight: '50px',
          textAlign: 'center',
          color: '#5079d9',
        }}
      >
        {text}
      </div>
      <Item>0</Item>
      <Item>1</Item>
      <Item>2</Item>
      <Item>3</Item>
      <Item>4</Item>
      <Item>5</Item>
      <Intersection
        onEnter={() => setText('Enter')}
        onLeave={() => setText('Leave')}
      />
      <Item>6</Item>
      <Item>7</Item>
      <Item>8</Item>
      <Item>9</Item>
      <Item>10</Item>
    </div>
  );
}
```
